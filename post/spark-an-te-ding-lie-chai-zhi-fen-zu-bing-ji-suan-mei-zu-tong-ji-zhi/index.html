Spark按特定列差值分组并计算每组统计值

2019-10-27
import operation.GroupTagFunction;
import org.apache.spark.sql.Column;
import org.apache.spark.sql.Dataset;
import org.apache.spark.sql.Encoders;
import org.apache.spark.sql.Row;
import org.apache.spark.sql.SQLContext;
import org.apache.spark.sql.SparkSession;
import org.apache.spark.sql.expressions.Window;
import org.apache.spark.sql.expressions.WindowSpec;
import org.apache.spark.sql.functions;
import org.apache.spark.sql.types.DataTypes;

import java.util.ArrayList;
import java.util.List;

import static org.apache.spark.sql.functions.avg;
import static org.apache.spark.sql.functions.first;
import static org.apache.spark.sql.functions.last;
import static org.apache.spark.sql.functions.max;

public class StatisticDemo {
    public static void main(String[] args) {
        // 初始化sparkSession local[*]表示CPU满载运行
        SparkSession sparkSession
                = SparkSession.builder().appName("spark demo").master("local[*]").getOrCreate();

        // 注册自定义函数，用于获取超过阈值时间差的分组tag
        SQLContext context = sparkSession.sqlContext();
        context.udf().register("groupTagFunction", new GroupTagFunction(), DataTypes.LongType);

        // 定义时间窗列
        WindowSpec windowSpec = Window.orderBy(new Column("time"));
        // 定义上一行时间戳列
        Column lagTimeStamp = functions.lag(new Column("time"), 1, 0.0).over(windowSpec);
        Column timeStamp = functions.to_timestamp(new Column("time"));
        Column timeDiff = timeStamp.minus(lagTimeStamp);

        // 定义源数据列
        List<String> textList = new ArrayList<String>() {{
            add("id,time,speed,mile");
            add("c1,2019-10-27 22:41:15,13,5000");
            add("c1,2019-10-27 22:41:25,18,5035");
            add("c1,2019-10-27 22:41:36,25,5075");
            add("c1,2019-10-27 22:41:47,32,5115");
            add("c1,2019-10-27 22:41:55,40,5155");
            add("c1,2019-10-27 22:42:04,41,5205");
            add("c1,2019-10-27 22:42:15,37,5255");
            add("c1,2019-10-27 22:42:25,33,5325");
            add("c1,2019-10-27 22:48:15,13,6000");
            add("c1,2019-10-27 22:48:25,18,6035");
            add("c1,2019-10-27 22:48:36,25,6075");
            add("c1,2019-10-27 22:48:47,32,6115");
            add("c1,2019-10-27 22:48:55,40,6155");
            add("c1,2019-10-27 22:49:04,41,6205");
            add("c1,2019-10-27 22:49:15,37,6255");
            add("c1,2019-10-27 22:49:25,33,6325");
        }};
        // 源数据读入
        Dataset<String> srcDataset = sparkSession.createDataset(textList, Encoders.STRING());

        // 新增上一时间戳列，当前时间戳列和两时间戳差值列，以及根据差值得到的tag列
        Dataset<Row> extDataset = srcDataset
                .withColumn("lagTimeStamp", lagTimeStamp)
                .withColumn("timeStamp", timeStamp)
                .withColumn("timeDiff", timeDiff)
                .withColumn("groupTag", functions.callUDF("groupTagFunction", timeDiff));

        // 按tag分组并统计每组信息
        Dataset<Row> destDataset = extDataset.groupBy("groupTag").
                agg(avg("speed").alias("avgSpeed"),
                        max("speed").alias("maxSpeed"),
                        first("mile").alias("startMile"),
                        last("mile").alias("endMile"),
                        last("mile").minus(first("mile").alias("range"))
                );
        destDataset.show();
    }
}
